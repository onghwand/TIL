## 5215. 햄버거 다이어트

```python
T = int(input())
for tc in range(1, T+1):
    N, L = map(int, input().split())
    arr = [list(map(int, input().split())) for _ in range(N)]
    maxV = 0
    
    #모든 부분집합 생성
    for i in range(1<<N): 
        lst = []
        for j in range(N):
            if i & (1<<j):
                lst.append(arr[j])
        # 부분집합에 대해 칼로리, 점수 총합
        calorie = 0
        score = 0
        for x in lst:
            score += x[0]
            calorie += x[1]
        
        # 칼로리 조건만족하고 최댓값보다 크면 갱신
        if calorie <= L and maxV < score:
            maxV = score
            
    print(f'#{tc} {maxV}')
```

```python
def f(s):
    if s == s[::-1]:
        return 'Exist'
    return 'Not exist'

for tc in range(1,T+1):

        tmp = []

                break
            else:
                tmp.append(s[i])
                tmp.append(s[-i-1])

            b = tmp.pop(0)

                break
        else:
            ans = 'Exist'

        ans = f(s)
```

## 3499. 퍼펙트 셔플

```python
T = int(input())
for tc in range(1, T+1):
    N = int(input())
    arr = list(input().split())
    if N % 2:
        pre = arr[:N // 2 + 1]
        post = arr[N // 2 + 1 :]
    else:
        pre = arr[:N // 2]
        post = arr[N // 2:]

    ans = [0]*N
    for i in range(len(pre)):
        ans[i*2] = pre[i]
    for j in range(len(post)):
        ans[j*2+1] = post[j]
```
